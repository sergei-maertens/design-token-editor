## Tokens block

Displays a context header with the relevant tokens contained inside.

export const Template = ({path, tokens}) => <TokensBlock path={path} tokens={tokens} />;

export const colorTokens = Object.values(ofDesignTokens.of.color).filter(
  potentialToken => 'value' in potentialToken
);

<Canvas>
  <Story name="Default" args={{path: ['of', 'color'], tokens: colorTokens}}>

  </Story>
